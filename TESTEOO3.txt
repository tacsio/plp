{
 
	module Teste {
	
		int constanteTeste = 4;
		
	} ,
	
	classe Contador  {
		include Teste;
		
		int valor = 0;
 
		Contador(int i){
			this.valor := i
		},
 
		proc print() {
			write(this.valor)
		}
 
	},
	
	classe ContadorFilho extends Contador {
		int valorFilho = 1;
		
		ContadorFilho(int i){
			this.valorFilho := i
		},
		
		proc hello() {
			write("hello")
 		}
 	};
 	
	{
		Contador cont := new ContadorFilho(1);
		write(cont.constanteTeste)
		
	}

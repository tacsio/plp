{
 
	module Teste {
		int constanteTeste = 2;
		
		proc printFromModule(int i){
			write(constanteTeste+i)
		}
	} ,
	
	classe Contador  {
		extends Teste;

		int valor = 0;
 
		Contador(int i){
			this.valor := i
		},
 
		proc print() {
			write(this.valor)
		}
 
	};
 	
	{
		Contador cont := new Contador(1);
		Contador.printFromModule(2);
		cont.printFromModule(2);
		write(cont.constanteTeste);
		write(Contador.constanteTeste)
	}
	
	
 
}

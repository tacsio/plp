{

	classe Contador  {

		extends modulo1, modulo2, modulo3;
		include modulo4, modulo5, modulo6;
		extends modulo2 , modulo7, modulo4;
		
		int valor = 0;

		Contador(int i){
			this.valor := i
		},

		proc print() {
			write(this.valor)
		}

	};

	{
		Contador cont := new Contador(1),
		Contador cont2 := new Contador(2);
		cont.print();
		cont2.print()
	}

}
